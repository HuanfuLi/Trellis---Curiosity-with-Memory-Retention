```typescript
function OptionChipGroup<T extends string>({
  label,
  options,
  selected,
  onSelect,
  labelFn,
}: {
  label: string;
  options: T[];
  selected: T;
  onSelect: (v: T) => void;
  labelFn: (v: T) => string;
}) {
  return (
    <div style={{ marginBottom: 12 }}>
      <div style={{ fontSize: '0.8rem', color: 'var(--muted-foreground)', marginBottom: 6, fontWeight: 600 }}>
        {label}
      </div>
      <div style={{ display: 'flex', gap: 8, flexWrap: 'wrap' }}>
        {options.map((opt) => (
          <button
            key={opt}
            onClick={() => onSelect(opt)}
            style={{
              padding: '6px 16px',
              borderRadius: 'var(--radius-xl)',
              border: 'none',
              fontSize: '0.85rem',
              fontWeight: 500,
              cursor: 'pointer',
              background: opt === selected ? 'var(--primary-40)' : 'var(--surface-variant)',
              color: opt === selected ? '#fff' : 'var(--on-surface)',
              transition: 'background 0.15s',
            }}
          >
            {labelFn(opt)}
          </button>
        ))}
      </div>
    </div>
  );
}
```
